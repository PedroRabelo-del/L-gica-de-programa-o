Exercício 1 – Calculadora (Estrutura escolha)
algoritmo "Calculadora"

var
   num1, num2, resultado: real
   opcao: inteiro

inicio

   escre("Digite o primeiro numero: ")
   leia(num1)

   escre("Digite o segundo numero: ")
   leia(num2)

   escreval("===== CALCULADORA =====")
   escreval("1 - Soma")
   escreval("2 - Subtracao")
   escreval("3 - Multiplicacao")
   escreval("4 - Divisao")
   escre("Escolha uma operacao: ")
   leia(opcao)

   escolha opcao

      caso 1
         resultado <- num1 + num2
         escreval("Resultado: ", resultado)

      caso 2
         resultado <- num1 - num2
         escreval("Resultado: ", resultado)

      caso 3
         resultado <- num1 * num2
         escreval("Resultado: ", resultado)

      caso 4
         // Impede divisao por zero
         se num2 <> 0 entao
            resultado <- num1 / num2
            escreval("Resultado: ", resultado:0:2)
         senao
            escreval("Erro: nao existe divisao por zero!")
         fimse

      outrocaso
         escreval("Opcao invalida!")

   fimescolha

fimalgoritmo
Exercício 2 – Tabuada (Estrutura para)
algoritmo "Tabuada"

var
   numero, contador, resultado: inteiro

inicio

   escre("Digite um numero inteiro positivo: ")
   leia(numero)

   escreval("Tabuada do ", numero)

   para contador de 1 ate 10 faca
      resultado <- numero * contador
      escreval(numero, " x ", contador, " = ", resultado)
   fimpara

Exercício 3 – Sistema de Login (Estrutura enquanto)
algoritmo "SistemaLogin"

var
   senhaCorreta, senhaDigitada: caractere

inicio

   senhaCorreta <- "senha123"

   escre("Digite a senha: ")
   leia(senhaDigitada)

   enquanto senhaDigitada <> senhaCorreta faca

      escreval("Senha incorreta, tente novamente.")
      escre("Digite a senha: ")
      leia(senhaDigitada)

   fimenquanto

   escreval("Acesso Liberado! Bem-vindo!")

fimalgoritmo
